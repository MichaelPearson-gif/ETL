Drop table crime;
CREATE TABLE crime (
  id INT PRIMARY KEY,
  reportyear int,
  geotype text,
  geoname text,
  county_name text,
  region_name text,
  strata_level_name text,
  no_of_crime int,	
  population int,
  rate INT
);
